## 📋 Data Format Specification Summary

Here's the core JSON Schema that can be directly copied for your reference:

```json
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "title": "Resonance-Missile Field Trajectory Schema",
  "description": "场域轨迹数据格式规范 - 工具理性悖论防护已内置",
  "type": "object",
  "properties": {
    "version": {"type": "string", "const": "1.0"},
    "metadata": {
      "type": "object",
      "properties": {
        "dataset_name": {"type": "string"},
        "total_steps": {"type": "integer", "minimum": 1},
        "created_at": {"type": "string", "format": "date-time"},
        "paradox_safeguard": {"type": "string", "const": "instrumental_paradox_protection_enabled"}
      },
      "required": ["dataset_name", "total_steps", "created_at", "paradox_safeguard"]
    },
    "trajectory": {
      "type": "array",
      "items": {
        "type": "object",
        "properties": {
          "step": {"type": "integer", "minimum": 1},
          "char_pos": {"type": "integer", "minimum": 0},
          "U": {"type": "number", "minimum": 0.0, "maximum": 1.0},
          "D": {"type": "number", "minimum": 0.0, "maximum": 1.0},
          "A": {"type": "number", "minimum": 0.0, "maximum": 1.0},
          "H": {"type": "number", "minimum": 0.0, "maximum": 1.0},
          "U_diff": {"type": "number"},
          "D_diff": {"type": "number"},
          "A_diff": {"type": "number"},
          "H_diff": {"type": "number"},
          "H_driven_by": {"type": "string", "enum": ["U", "D", "A", "均衡"]},
          "cumul_A": {"type": "number", "minimum": 0.0},
          "D_volatility": {"type": "number", "minimum": 0.0}
        },
        "required": ["step", "char_pos", "U", "D", "A", "H"]
      }
    }
  },
  "required": ["version", "metadata", "trajectory"]
}
```
